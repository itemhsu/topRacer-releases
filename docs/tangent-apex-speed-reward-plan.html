def braking_zone(entries, n):
    """煞車區：每個入口為中心連續 6 點 e-3..e+2。"""
    z = set()
    for e in entries:
        for k in range(-3, 3):
            z.add((e + k) % n)
    return z

STRAIGHT_SPEED, CORNER_SPEED, BRAKE_SPEED = 4.0, 2.5, 2.0   # 對齊動作空間 speed_high
TOP = STRAIGHT_SPEED          # 速度項正規化分母（出彎也用此式）
W_STEER = W_SPEED = 1.0       # A 案：相加


def _clip01(x):
    return 0.0 if x < 0.0 else (1.0 if x > 1.0 else x)


def speed_term(speed, i, c):
    """區域化連續速度項 v∈[0,1]。優先序 煞車 > 出彎 > 彎中 > 直線。"""
    if i in c["brake"]:
        return _clip01(1.0 - abs(speed - BRAKE_SPEED) / TOP)    # @2.0
    if i in c["exit"]:
        return _clip01(speed / STRAIGHT_SPEED)                  # 後三點全速出彎
    if i in c["left"] or i in c["right"]:
        return _clip01(1.0 - abs(speed - CORNER_SPEED) / TOP)   # @2.5
    return _clip01(speed / STRAIGHT_SPEED)                      # 直線

def reward_function(params):
    """切點轉向（0/1）＋ 區間速度（連續）相加。"""
    global _CACHE
    if _CACHE is None:
        _CACHE = build_cache(params["waypoints"], params["track_width"])
    c = _CACHE
    steering = float(params["steering_angle"])
    speed = float(params["speed"])
    _, nxt = params["closest_waypoints"]
    corner = next_corner(int(nxt), c["corners"], c["n"])
    if corner is None:
        steer = reward_for(0.0, steering)
    else:
        P = (float(params["x"]), float(params["y"]))
        T = tangent_point(P, apex_polyline(corner, c["apex"], c["n"]), corner[2] == "L")
        e = normalize_deg(bearing(P, T) - float(params["heading"]))
        steer = reward_for(e, steering)
    v = speed_term(speed, int(nxt), c)
    return W_STEER * steer + W_SPEED * v

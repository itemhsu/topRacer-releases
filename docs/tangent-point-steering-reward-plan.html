"""切點行為獎勵 (Tangent-Point Steering Reward)."""
import math

TOL_ALIGN = 6.0       # deg: |e| <= this => aligned (should go straight)
TOL_STRAIGHT = 3.0    # deg: |steering| <= this => "straight" (not turning)
EPS = 0.15            # turn3 curvature threshold for bend detection

_CACHE = None         # built once on first call (centreline is static)


def turn3(a, b, c):
    """Signed, length-normalised curvature at b (left > 0)."""
    v1x, v1y = b[0] - a[0], b[1] - a[1]
    v2x, v2y = c[0] - b[0], c[1] - b[1]
    cross = v1x * v2y - v1y * v2x
    scale = (math.hypot(v1x, v1y) * math.hypot(v2x, v2y)) or 1.0
    return cross / scale


def classify_bends(wp, n, eps=EPS):
    """Return (left, right) index sets by signed curvature over the loop."""
    left, right = set(), set()
    for i in range(n):
        t = turn3(wp[(i - 2) % n], wp[i % n], wp[(i + 2) % n])
        if t > eps:
            left.add(i)
        elif t < -eps:
            right.add(i)
    return left, right


def find_entries(bend, n):
    """Corner entry indices: in a bend, previous index not in a bend."""
    return [i for i in range(n) if i in bend and (i - 1) % n not in bend]

def left_normal(wp, i, n):
    """Unit left normal of the local travel direction at index i (closed loop)."""
    tx = wp[(i + 1) % n][0] - wp[(i - 1) % n][0]
    ty = wp[(i + 1) % n][1] - wp[(i - 1) % n][1]
    L = math.hypot(tx, ty) or 1.0
    return (-ty / L, tx / L)


def corners_from(left, right, n):
    """Group consecutive same-direction bend indices into (start, end, dir)."""
    cls = ['L' if i in left else ('R' if i in right else '.') for i in range(n)]
    segs, i = [], 0
    while i < n:
        if cls[i] in 'LR':
            d = cls[i]
            j = i
            while j < n and cls[j] == d:
                j += 1
            segs.append((i, j - 1, d))
            i = j
        else:
            i += 1
    return segs


def build_cache(waypoints, track_width, eps=EPS):
    """Four tables once: bend sets, entries, corner segments, apex-side points."""
    wp = [(float(x), float(y)) for x, y in waypoints]
    n = len(wp)
    left, right = classify_bends(wp, n, eps)
    entries = find_entries(left | right, n)
    corners = corners_from(left, right, n)
    half = float(track_width) / 2.0
    apex = {}
    for s, e, d in corners:
        for i in range(s, e + 1):
            nx, ny = left_normal(wp, i, n)
            sgn = 1.0 if d == 'L' else -1.0       # apex = inside of the turn
            apex[i] = (wp[i][0] + sgn * half * nx, wp[i][1] + sgn * half * ny)
    return {"left": left, "right": right, "entries": entries,
            "corners": corners, "apex": apex, "n": n}

def next_corner(next_idx, corners, n):
    """The corner containing next_idx, else the nearest ahead; None if empty."""
    if not corners:
        return None
    best, best_gap = None, None
    for (s, e, d) in corners:
        within = s <= next_idx <= e
        gap = 0 if within else (s - next_idx) % n
        if best_gap is None or gap < best_gap:
            best_gap, best = gap, (s, e, d)
    return best


def apex_polyline(corner, apex, n):
    """The apex-side points along a corner segment, in index order."""
    s, e, _ = corner
    return [apex[i] for i in range(s, e + 1)]

def cross(p, a, b):
    """(a-p) x (b-p): > 0 means b is to the left of the ray p->a."""
    return (a[0] - p[0]) * (b[1] - p[1]) - (a[1] - p[1]) * (b[0] - p[0])


def tangent_point(P, B, left_turn):
    """Supporting point of polyline B seen from P: the whole of B lies on one
    side of the sight-line P->point. Left turn -> 'all others left' extreme."""
    for cand in B:
        ok = all((cross(P, cand, o) >= 0) == left_turn
                 for o in B if o is not cand)
        if ok:
            return cand
    return B[len(B) // 2]      # degenerate (collinear/noisy) -> midpoint

def normalize_deg(a):
    """Wrap an angle in degrees to the half-open range (-180, 180]."""
    a = (a + 180.0) % 360.0 - 180.0
    return a + 360.0 if a <= -180.0 else a


def bearing(P, T):
    """Bearing from P to T in degrees (same frame as DeepRacer heading)."""
    return math.degrees(math.atan2(T[1] - P[1], T[0] - P[0]))

def reward_for(e, steering, tol_align=TOL_ALIGN, tol_straight=TOL_STRAIGHT):
    """Binary reward (spec section 5): 1.0 iff the steering command is the
    correct corrective action for alignment error e (+ = tangent to the left)."""
    if e > tol_align:                                  # tangent left -> turn LEFT
        return 1.0 if steering > tol_straight else 0.0
    if e < -tol_align:                                 # tangent right -> turn RIGHT
        return 1.0 if steering < -tol_straight else 0.0
    return 1.0 if abs(steering) <= tol_straight else 0.0   # aligned -> go straight

def reward_function(params):
    """DeepRacer entry point. Reward correct corrective steering toward the next
    corner's dynamic tangent point; build & cache the bend tables on first call."""
    global _CACHE
    if _CACHE is None:
        _CACHE = build_cache(params["waypoints"], params["track_width"])
    c = _CACHE
    steering = float(params["steering_angle"])
    _, nxt = params["closest_waypoints"]
    corner = next_corner(int(nxt), c["corners"], c["n"])
    if corner is None:
        return reward_for(0.0, steering)               # no corner -> reward straight
    P = (float(params["x"]), float(params["y"]))
    B = apex_polyline(corner, c["apex"], c["n"])
    T = tangent_point(P, B, corner[2] == "L")
    e = normalize_deg(bearing(P, T) - float(params["heading"]))
    return reward_for(e, steering)

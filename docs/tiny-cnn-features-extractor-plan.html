import torch
import torch.nn as nn
from gymnasium import spaces
from stable_baselines3.common.torch_layers import BaseFeaturesExtractor


def _dwsep(cin, cout, stride):
    """Depthwise-separable block: depthwise 3x3 (per-channel) + pointwise 1x1 + ReLU."""
    return nn.Sequential(
        nn.Conv2d(cin, cin, 3, stride=stride, padding=1, groups=cin, bias=False),
        nn.Conv2d(cin, cout, 1, bias=False),
        nn.ReLU(),
    )


class TinyCNN(BaseFeaturesExtractor):
    """Tiny CNN features extractor for the PPO policy. Keeps the input resolution but
    shrinks the network ~200x vs NatureCNN: a large-stride conv1 to compress space
    fast, two depthwise-separable blocks, Global Average Pooling (which drops the huge
    flatten->FC), then a small FC. Input is a channels-first uint8 image (SB3
    transposes HWC->CHW); it is normalised to [0,1] inside forward()."""

    def __init__(self, observation_space: spaces.Box, features_dim: int = 128):
        super().__init__(observation_space, features_dim)
        c = observation_space.shape[0]
        self.cnn = nn.Sequential(
            nn.Conv2d(c, 16, 8, stride=4), nn.ReLU(),   # big stride -> fast spatial compress
            _dwsep(16, 32, stride=2),
            _dwsep(32, 32, stride=1),
            nn.AdaptiveAvgPool2d(1),                    # GAP -> (B, 32, 1, 1)
            nn.Flatten(),                               # -> (B, 32)
        )
        self.fc = nn.Sequential(nn.Linear(32, features_dim), nn.ReLU())

    def forward(self, observations: torch.Tensor) -> torch.Tensor:
        return self.fc(self.cnn(observations.float() / 255.0))

def _circle(n=60, R=10.0, hw=1.0):
    th = np.linspace(0, 2 * np.pi, n, endpoint=False)
    def ring(r): return np.stack([r * np.cos(th), r * np.sin(th)], axis=1)
    return np.concatenate([ring(R), ring(R - hw), ring(R + hw)], axis=1)

def build_model(env_fn, cfg, ppo_kwargs=None):
    ppo_kwargs = dict(ppo_kwargs or {})
    vec = DummyVecEnv([env_fn for _ in range(cfg.n_envs)])
    device = select_train_device()
    policy_kwargs = None
    if getattr(cfg, "features_extractor", "nature") == "tiny":
        from topracer.train.tiny_cnn import TinyCNN
        policy_kwargs = {"features_extractor_class": TinyCNN,
                         "features_extractor_kwargs": {"features_dim": 128}}
    return EpisodicPPO("CnnPolicy", vec, device=device, verbose=0,
                       policy_kwargs=policy_kwargs, **ppo_kwargs)

def count_params(module) -> int:
    """Total trainable parameter count of a torch module."""
    return sum(p.numel() for p in module.parameters())


def compare_param_counts(observation_space, features_dim: int = 128) -> dict:
    """Param counts of TinyCNN vs SB3's default NatureCNN on the same obs space."""
    from stable_baselines3.common.torch_layers import NatureCNN
    return {"tiny": count_params(TinyCNN(observation_space, features_dim)),
            "nature": count_params(NatureCNN(observation_space, features_dim))}

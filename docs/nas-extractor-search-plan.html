本計畫先建好可測試的基礎建設(抽取器家族、登錄表、sweep 指標、彙整/挑選/報告);真正的 36-變體訓練 sweep 是獨立的重算步驟(見最後),由你決定何時開跑。

def _block(cin, cout, stride, conv):
    """One downsample block: 'full' = Conv3x3, 'dw' = depthwise-separable; both + ReLU."""
    if conv == "full":
        return nn.Sequential(nn.Conv2d(cin, cout, 3, stride=stride, padding=1), nn.ReLU())
    return _dwsep(cin, cout, stride)


class ConfigurableCNN(BaseFeaturesExtractor):
    """Parametrised small CNN for the NAS sweep: conv1 (k8, s4) + stride-2 blocks
    (depthwise-separable or full) -> Flatten (keeps spatial) -> FC. channels =
    (conv1_out, block1, block2, ...); #stride-2 blocks (len(channels)-1) sets the grid."""

    def __init__(self, observation_space: spaces.Box, channels=(16, 32, 32, 32),
                 fc_dim: int = 128, conv: str = "dw"):
        super().__init__(observation_space, fc_dim)
        c = observation_space.shape[0]
        layers = [nn.Conv2d(c, channels[0], 8, stride=4), nn.ReLU()]
        for cin, cout in zip(channels[:-1], channels[1:]):
            layers.append(_block(cin, cout, 2, conv))
        layers.append(nn.Flatten())
        self.cnn = nn.Sequential(*layers)
        with torch.no_grad():
            n = self.cnn(torch.zeros(1, *observation_space.shape)).shape[1]
        self.fc = nn.Sequential(nn.Linear(n, fc_dim), nn.ReLU())

    def forward(self, observations: torch.Tensor) -> torch.Tensor:
        return self.fc(self.cnn(observations))

_REGISTRY = {}


def register_variant(name, cls, **kwargs):
    """Register a features-extractor variant so build_model can select it by name."""
    _REGISTRY[name] = (cls, kwargs)


def make_policy_kwargs(name):
    """policy_kwargs for a features_extractor name; None for 'nature' (SB3 default)."""
    if name == "nature":
        return None
    if name == "tiny":
        return {"features_extractor_class": TinyCNN}
    if name in _REGISTRY:
        cls, kw = _REGISTRY[name]
        return {"features_extractor_class": cls, "features_extractor_kwargs": kw}
    return None

# (2) topracer/train/train.py — replace build_model's extractor branch with:
def build_model(env_fn, cfg, ppo_kwargs=None):
    ppo_kwargs = dict(ppo_kwargs or {})
    vec = DummyVecEnv([env_fn for _ in range(cfg.n_envs)])
    device = select_train_device()
    from topracer.train.tiny_cnn import make_policy_kwargs
    pk = make_policy_kwargs(getattr(cfg, "features_extractor", "nature"))
    if pk is not None:
        ppo_kwargs["policy_kwargs"] = pk
    return EpisodicPPO("CnnPolicy", vec, device=device, verbose=0, **ppo_kwargs)

import itertools

from topracer.train.tiny_cnn import ConfigurableCNN, register_variant

WIDTHS = {"w16": (16, 32, 32), "w24": (24, 48, 48), "w32": (32, 64, 64)}
GRIDS = {"g8x10": 0, "g4x5": 1}     # 1 = one extra stride-2 block -> smaller grid
FCS = (96, 128, 192)
CONVS = ("dw", "full")


def sweep_variants():
    """The full 36-cell grid as [(name, kwargs)], each registered by name."""
    out = []
    for (wn, w), (gn, extra), fc, conv in itertools.product(
            WIDTHS.items(), GRIDS.items(), FCS, CONVS):
        channels = tuple(w) + (w[-1],) * extra
        name = f"{wn}_{gn}_fc{fc}_{conv}"
        kw = dict(channels=channels, fc_dim=fc, conv=conv)
        register_variant(name, ConfigurableCNN, **kw)
        out.append((name, kw))
    return out


def convergence_iter(training_json, thresh=95.0):
    """First iteration whose eval_progress >= thresh; None if it never reaches it."""
    for it in training_json.get("iterations", []):
        if (it.get("eval_progress") or 0) >= thresh:
            return it.get("iteration")
    return None

import statistics


def aggregate(results, params):
    """results: {name: [conv_iter|None per seed]}; params: {name: int}.
    -> {name: {"median": float|None, "reached": bool, "params": int}}."""
    agg = {}
    for name, iters in results.items():
        ok = [i for i in iters if i is not None]
        agg[name] = {"median": statistics.median(ok) if ok else None,
                     "reached": len(iters) > 0 and len(ok) == len(iters),
                     "params": params.get(name)}
    return agg


def pick_smallest(agg, baseline_iter, factor=1.5):
    """Smallest-params variant that reached eval95 on every seed and whose median
    convergence iter <= baseline_iter*factor. None if nothing qualifies."""
    cap = baseline_iter * factor
    elig = [(v["params"], n) for n, v in agg.items()
            if v["reached"] and v["median"] is not None and v["median"] <= cap]
    return min(elig)[1] if elig else None


def render_report(agg, baseline_iter, pick, out_path):
    """Write an HTML table of every variant (params / median conv iter / reached) and
    highlight the recommended pick. Returns out_path."""
    rows = sorted(agg.items(), key=lambda kv: (kv[1]["params"] or 0))
    trs = "".join(
        f"<tr style='background:{'#10261a' if n == pick else 'transparent'}'>"
        f"<td>{n}</td><td>{v['params']}</td><td>{v['median']}</td>"
        f"<td>{'OK' if v['reached'] else 'x'}</td></tr>" for n, v in rows)
    doc = (f"<!doctype html><meta charset=utf-8><title>NAS sweep</title>"
           f"<h1>NAS sweep (baseline conv={baseline_iter}, cap={baseline_iter * 1.5})</h1>"
           f"<p><b>推薦(最小且達標):</b> {pick}</p>"
           f"<table border=1><tr><th>variant</th><th>params</th>"
           f"<th>median conv iter</th><th>reached95</th></tr>{trs}</table>")
    open(out_path, "w", encoding="utf-8").write(doc)
    return out_path

def run_one(name, seed, iters, track, action, reward_src, root="models/nas"):
    cfg_dir = f"{root}/cfg_{name}_s{seed}"
    build_config_dir(cfg_dir, track_name=track, action_space=action, reward_src=reward_src,
